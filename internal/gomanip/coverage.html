package gomanip

import (
        "bytes"
        "context"
        "encoding/json"
        "errors"

        "time"

        "github.com/cenkalti/backoff/v5"

)

var (
        apiError     = errors.New("error returned from api")
        serverError  = errors.New("server side error")
        networkError = errors.New("netork error")
        readError    = errors.New("error marshaling response body")
        retryError   = errors.New("retrying endpoint")
)

type ErrorResponse struct {
        Detail string `json:"detail"`
}

type GoManip struct {
        apiEndpoint string
}

func NewGoManip(apiEndpoint string) *GoManip {
        return &GoManip{
                apiEndpoint: apiEndpoint,
        }
}

func (g *GoManip) tryApi(apiURI, contentType string, imageBytesBuffer *bytes.Buffer) (*http.Response, error) {
        client := http.Client{}
        resp, err := client.Post(apiURI, contentType, imageBytesBuffer)
        if err != nil {
                return nil, backoff.Permanent(fmt.Errorf("error sending request: %v; %w", err, networkError))

        }

        if resp.StatusCode == http.StatusOK {
                return resp, nil
        }

        var errorResponse ErrorResponse
        body, err := io.ReadAll(resp.Body)
        if err != nil {
                return nil, backoff.Permanent(fmt.Errorf("error reading response body: %v; %w", err, readError))
        }

        err = json.Unmarshal(body, &errorResponse)
        if err != nil {
                return nil, backoff.Permanent(fmt.Errorf("failed to unmarshal response body: %v; %w", err, readError))
        }

        if resp.StatusCode == http.StatusBadRequest {
                return nil, backoff.Permanent(fmt.Errorf("%s; %w", errorResponse.Detail, apiError))
        }

        if resp.StatusCode >= 500 {
                return nil, backoff.Permanent(fmt.Errorf("server reported error in response: %s; %w", errorResponse.Detail, serverError))
        }

        return nil, fmt.Errorf("status not OK, got: %s. %w", errorResponse.Detail, retryError)
}

func (g *GoManip) Do(image []byte, contentType, endpoint, queries string) ([]byte, error) {
        apiURI := fmt.Sprintf("%s/%s/%s", g.apiEndpoint, endpoint, queries)

        var imageBytesBuffer *bytes.Buffer

        maxDuration := 30 * time.Second
        timeoutCtx, cancel := context.WithTimeout(context.Background(), maxDuration)

        defer cancel()

        resp, err := backoff.Retry(
                timeoutCtx,
                func() (*http.Response, error) {
                        imageBytesBuffer = bytes.NewBuffer(image)
                        return g.tryApi(apiURI, contentType, imageBytesBuffer)
                },
                backoff.WithBackOff(backoff.NewExponentialBackOff()),
        )

        if err != nil {
                log.Err(err).Msg("error calling gomanip service")
                return nil, err
        }

        defer resp.Body.Close()

        resultBytes, err := io.ReadAll(resp.Body)
        if err != nil {
                log.Err(err).Msg("failed reading response body")
                return nil, fmt.Errorf("failed reading response body; %w", readError)
        }

        return resultBytes, nil

}

var (
        ErrTimedOut  = errors.New("the service timed out")
        ErrBadParams = errors.New("given parameters are invalid")
        ErrGeneral   = errors.New("error while calling api")
)

func errorChecker(err error) error {
        if errors.Is(err, retryError) {
                return fmt.Errorf("request timed out. %w", err)
        }
        if errors.Is(err, apiError) {
                return fmt.Errorf("%v, %w", err, ErrBadParams)
        }
        if err != nil {
                return fmt.Errorf("failed to get image from api: %v, %w", err, ErrGeneral)
        }
        return nil
}
